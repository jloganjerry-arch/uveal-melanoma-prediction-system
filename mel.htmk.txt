    <script>
        // DOM Elements
        const loginView = document.getElementById('login-view');
        const dashboardView = document.getElementById('dashboard-view');
        const loginForm = document.getElementById('login-form');
        const predictionForm = document.getElementById('prediction-form');
        const userRoleDisplay = document.getElementById('user-role-display');
        
        const resultPlaceholder = document.getElementById('result-placeholder');
        const resultDisplay = document.getElementById('result-display');
        const scoreOutput = document.getElementById('score-output');
        const categoryOutput = document.getElementById('category-output');

        // --- 1. Handle Login ---
        loginForm.addEventListener('submit', function(e) {
            e.preventDefault(); // Stop actual page reload
            
            const role = document.getElementById('role').value;
            
            // Update UI with selected role
            userRoleDisplay.textContent = role.charAt(0).toUpperCase() + role.slice(1);

            // Switch Views
            loginView.classList.add('hidden');
            dashboardView.classList.remove('hidden');
        });

        // --- 2. Handle Logout ---
        function handleLogout() {
            // Reset Forms
            loginForm.reset();
            predictionForm.reset();
            
            // Reset Results View
            resultDisplay.classList.add('hidden');
            resultPlaceholder.classList.remove('hidden');

            // Switch Views
            dashboardView.classList.add('hidden');
            loginView.classList.remove('hidden');
        }

        // --- 3. Handle Prediction Simulation ---
        predictionForm.addEventListener('submit', function(e) {
            e.preventDefault(); // Stop actual form submission to backend

            // In a real Python/Django app, you would use fetch() here.
            // Since this is a standalone HTML demo, we will simulate the result.

            // Simulate loading state (optional polish)
            const btn = predictionForm.querySelector('button');
            const originalText = btn.innerText;
            btn.innerText = "Processing...";
            btn.disabled = true;

            setTimeout(() => {
                // Generate a dummy random score between 10 and 90
                const randomScore = Math.floor(Math.random() * (90 - 10 + 1) + 10);
                
                // Determine category based on score
                let riskCategory = "Low Risk";
                let color = "#10b981"; // Green

                if(randomScore > 40) {
                    riskCategory = "Intermediate Risk";
                    color = "#f59e0b"; // Amber
                }
                if(randomScore > 70) {
                    riskCategory = "High Risk";
                    color = "#ef4444"; // Red
                }

                // Update UI
                scoreOutput.innerText = randomScore + "%";
                scoreOutput.style.color = color;
                categoryOutput.innerText = riskCategory;
                categoryOutput.style.color = color;

                // Show results, hide placeholder
                resultPlaceholder.classList.add('hidden');
                resultDisplay.classList.remove('hidden');
                resultDisplay.classList.add('fade-in');

                // Reset Button
                btn.innerText = originalText;
                btn.disabled = false;

            }, 800); // 800ms simulated delay
        });
    </script>
